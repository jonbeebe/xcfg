    public constructor(id: string, options: Object) source

    Creates a new Xcfg instance, initializes config directory structure and prepares a blank config file if it doesn't exist.

    Params:

    NameTypeAttributeDescription
    id string

    Identifier (used to create config directory in ~/.config), sanitized with: id.replace(/[/?<>\\:*|" :]/g, '.').replace(/\.+/g, '.')
    options Object

    options to change default behavior.
    options.confdir string

    override directory name (default: '.config')
    options.dir_mode number

    override directory file mode (default: parseInt('0700', 8)).
    options.file_mode number

    override file mode (default: parseInt('0600', 8)).
    options.filename string

    override file name (default: 'config.json').
    options.minify boolean

    when saving config, minify the output (default: false).
